1.Implementation Overview

This document is the definitive build guide for the Autonomous AI Job Application Platform — a 20-agent, multi-phase system purpose-built for Vikram Deshpande's Melbourne-based Technical Delivery Lead and AI Solutions Architect job search. It is a direct companion to the Architecture Document and translates every architectural decision into executable steps, code artifacts, and configuration files. An AI coding agent (Cursor, Aider, Devin) or a human development team can follow this guide sequentially from environment bootstrap to production deployment without requiring external clarification.

1.1System Summary

The platform is a full-stack TypeScript/Python monorepo that automates the entire job application lifecycle: discovery → evaluation → resume tailoring → cover letter generation → form submission → inbox management → outcome tracking → strategy adjustment. It is grounded in Vikram's demonstrated track record — 92% automation efficiency gains at ATO, $5M+ portfolio management at ANZ, 38% LLM error-rate reduction — and targets AI Architect, Senior Delivery Lead, and Technical Product Owner roles across ANZ, NAB, Telstra, and government agencies in Australia.

2.4.1NextAuth.js Configuration

// apps/web/app/api/auth/[...nextauth]/route.ts
import NextAuth from 'next-auth'
import GoogleProvider from 'next-auth/providers/google'
import GitHubProvider from 'next-auth/providers/github'
import LinkedInProvider from 'next-auth/providers/linkedin'
import CredentialsProvider from 'next-auth/providers/credentials'

export const authOptions = {
  providers: [
    GoogleProvider({ clientId: process.env.GOOGLE_CLIENT_ID!, clientSecret: process.env.GOOGLE_CLIENT_SECRET! }),
    GitHubProvider({ clientId: process.env.GITHUB_CLIENT_ID!, clientSecret: process.env.GITHUB_CLIENT_SECRET! }),
    LinkedInProvider({ clientId: process.env.LINKEDIN_CLIENT_ID!, clientSecret: process.env.LINKEDIN_CLIENT_SECRET! }),
    CredentialsProvider({
      name: 'credentials',
      credentials: { email: { type: 'email' }, password: { type: 'password' } },
      async authorize(credentials) {
        const res = await fetch(`${process.env.API_URL}/api/auth/login`, {
          method: 'POST', body: JSON.stringify(credentials),
          headers: { 'Content-Type': 'application/json' }
        })
        const data = await res.json()
        return res.ok ? data.user : null
      }
    })
  ],
  session: { strategy: 'jwt', maxAge: 30 * 24 * 60 * 60 },
  callbacks: {
    async jwt({ token, user }) {
      if (user) { token.userId = user.id; token.role = user.role }
      return token
    },
    async session({ session, token }) {
      session.user.id = token.userId as string
      return session
    }
  }
}

# packages/agents/graphs/base_graph.py
from langgraph.graph import StateGraph, END
from langgraph.checkpoint.postgres import PostgresSaver
from langfuse.langchain import CallbackHandler
import os

def build_graph(state_class, nodes: dict, edges: list, checkpoints=True):
    graph = StateGraph(state_class)
    for name, fn in nodes.items():
        graph.add_node(name, fn)
    for src, dst, condition in edges:
        if condition:
            graph.add_conditional_edges(src, condition)
        else:
            graph.add_edge(src, dst)
    checkpointer = PostgresSaver.from_conn_string(
        os.environ["DATABASE_URL"]
    ) if checkpoints else None
    return graph.compile(checkpointer=checkpointer, interrupt_before=["human_review"])

3.1.4Error Handling Pattern

# Standard error handling for all agent nodes
import functools
from tenacity import retry, stop_after_attempt, wait_exponential

def agent_node(name: str):
    def decorator(fn):
        @retry(stop=stop_after_attempt(3), wait=wait_exponential(min=1, max=10))
        @functools.wraps(fn)
        async def wrapper(state: AgentState) -> AgentState:
            try:
                result = await fn(state)
                await log_agent_action(name, state["trace_id"], result)
                return result
            except Exception as e:
                await log_error(name, state, str(e))
                return {**state, "error": str(e), "current_step": "failed"}
        return wrapper
    return decorator

# ATS scoring algorithm
def calculate_ats_score(resume_text: str, job_description: str) -> dict:
    # 1. Hard keyword matching (exact + stemmed)
    jd_keywords   = extract_keywords(job_description, top_n=50)
    resume_terms  = extract_keywords(resume_text,    top_n=100)
    keyword_score = jaccard_similarity(jd_keywords, resume_terms) * 100

    # 2. Semantic similarity (pgvector cosine distance)
    jd_vec        = embed_text(job_description)
    resume_vec    = embed_text(resume_text)
    semantic_score = (1 - cosine_distance(jd_vec, resume_vec)) * 100

    # 3. Experience level match
    exp_score = evaluate_experience_match(
        resume_years=15, required_years=parse_years_required(job_description)
    ) * 100

    # Weighted final score
    final = (0.45 * keyword_score + 0.35 * semantic_score + 0.20 * exp_score)
    return { "score": round(final, 1), "keyword": keyword_score,
             "semantic": semantic_score, "experience": exp_score,
             "missing_keywords": list(jd_keywords - resume_terms) }

from langgraph.graph import StateGraph, END
from .agents import *

def build_orchestration_graph():
    graph = StateGraph(AgentState)

    # Nodes
    graph.add_node("evaluate",         evaluation_agent)
    graph.add_node("research_company",  company_research_agent)
    graph.add_node("tailor_resume",    resume_tailoring_agent)
    graph.add_node("generate_cl",      cover_letter_agent)
    graph.add_node("quality_check",    quality_agent)
    graph.add_node("human_review",     human_review_node)    # interrupt point
    graph.add_node("detect_form",      form_detection_agent)
    graph.add_node("submit",           submission_agent)
    graph.add_node("track_outcome",    outcome_tracking_agent)

    # Entry point
    graph.set_entry_point("evaluate")

    # Conditional routing after evaluation
    def route_after_evaluation(state):
        level = state["confidence_level"]
        if level == "AUTO_APPLY":   return "research_company"
        if level == "NEEDS_APPROVAL": return "human_review"
        return END  # SKIP

    graph.add_conditional_edges("evaluate", route_after_evaluation,
        { "research_company": "research_company",
          "human_review": "human_review", END: END })

    # Sequential pipeline: research → tailor → cover letter → quality
    graph.add_edge("research_company", "tailor_resume")
    graph.add_edge("tailor_resume",    "generate_cl")
    graph.add_edge("generate_cl",      "quality_check")

    # Quality gate routing
    def route_after_quality(state):
        if state["quality_passed"] and state["confidence_level"] == "AUTO_APPLY":
            return "detect_form"
        return "human_review"   # anything else needs eyes on it

    graph.add_conditional_edges("quality_check", route_after_quality,
        { "detect_form": "detect_form", "human_review": "human_review" })

    # After human approval, resume at detect_form
    graph.add_edge("human_review",    "detect_form")
    graph.add_edge("detect_form",     "submit")
    graph.add_edge("submit",           "track_outcome")
    graph.add_edge("track_outcome",   END)

    return graph.compile(
        checkpointer=PostgresSaver.from_conn_string(os.environ["DATABASE_URL"]),
        interrupt_before=["human_review"]
    )

                viewport={"width": 1366, "height": 768},
                locale="en-AU"
            )
            page = await context.new_page()
            await page.goto(self._build_url(query, location))
            jobs = []
            while len(jobs) < max_results:
                items = await page.query_selector_all(".job-search-card")
                for item in items[:max_results - len(jobs)]:
                    job = await self._parse_card(item)
                    if job: jobs.append(job)
                if not await self._has_next_page(page): break
                await asyncio.sleep(random.uniform(self.MIN_DELAY, self.MAX_DELAY))
                await self._click_next(page)
            await browser.close()
            return jobs

5.1.2Strategy Adjustment Algorithm

# packages/agents/services/strategy.py
from scipy import stats
import numpy as np

def evaluate_ats_threshold(outcomes: list[LearningOutcome]) -> StrategyRecommendation:
    high_ats = [o for o in outcomes if o.features["ats_score"] >= 80]
    low_ats  = [o for o in outcomes if 70 <= o.features["ats_score"] < 80]

    rate_high = np.mean([o.interview_rate for o in high_ats]) if high_ats else 0
    rate_low  = np.mean([o.interview_rate for o in low_ats]) if low_ats  else 0

    _, p_value = stats.proportions_ztest(
        [sum(o.interview_rate for o in high_ats), sum(o.interview_rate for o in low_ats)],
        [len(high_ats), len(low_ats)]
    )
    if p_value < 0.05 and rate_high > rate_low * 1.2:
        return StrategyRecommendation(
            parameter="min_ats_score", current=70, recommended=75,
            evidence=f"High-ATS rate={rate_high:.2%} vs Low-ATS={rate_low:.2%}, p={p_value:.4f}"
        )
